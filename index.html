Enthält die Geschäftslogik u.a. Validierung und Steuerung der Oberfläche.

View

Enthält die Gestaltung der Oberfläche. 

<?xml version="1.0" encoding="utf-8" ?>
<ContentPage xmlns="http://xamarin.com/schemas/2014/forms"
        xmlns:x="http://schemas.microsoft.com/winfx/2009/xaml"
        x:Class="POI.Client.Settings">
    <ContentPage.Content>
        <StackLayout>
            <Label Text="Benutzer"></Label>
            <Entry Placeholder="Vorname und Nachname" Text="{Binding User, Mode=TwoWay}"></Entry>
            <Label Text="Service-URL"></Label>
            <Entry Placeholder="komplette URL" Text="{Binding ServiceUrl, Mode=TwoWay}"></Entry>
            <Button Text="Speichern" Command="{Binding SaveCommand}"></Button>
        </StackLayout>
    </ContentPage.Content>
</ContentPage>
